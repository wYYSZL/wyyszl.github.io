#include<bits/stdc++.h>

using namespace std;

char a[2005];
char b[2005];
int f[2005][2005];
int main(){
	ios::sync_with_stdio(0);
	scanf("%s",a+1);
	scanf("%s",b+1);
	int lena=strlen(a+1);
	int lenb=strlen(b+1);
	for(int i=1;i<=lena;++i){
		for(int j=1;j<=lenb;++j){
			if(not(a[i]-b[j])){
				f[i][j]=f[i-1][j-1]+1;
			}else {
				f[i][j]=max(f[i-1][j],f[i][j-1]);
			}
		}
	}
	cout<<max(lena,lenb)-f[lena][lenb];
	return 0;
}
