C++常用模版

#include<bits/stdc++.h>

using namespace std;

class s_r {//快读
	private:
	public:
		inline int read();
};
class s_w {//快写
	private:
	public:
		inline void write(int X);
};
class dfss {//DFS
	private:
		int n;
		int a[25];
		int b[25];
		void dfs(int t);
	public:
		void main();
};
inline int s_r::read() {
	int x=0,f=1;
	char ch=getchar();
	while(ch<'0'||ch>'9') {
		if(ch=='-')
			f=-1;
		ch=getchar();
	}
	while(ch>='0'&&ch<='9') {
		x=(x<<1)+(x<<3)+(ch^48);
		ch=getchar();
	}
	return x*f;
}
inline void s_w::write(int X) {
	if(X<0) {
		X=~(X-1);
		putchar('-');
	}
	if(X>9) write(X/10);
	putchar(X%10+'0');
	return;
}
void dfss::dfs(int t) {
	if(t==n+1) {
		for(int i=1; i<=n; ++i) {
			cout<<a[i]<<' ';
		}
		cout<<endl;
		return ;
	}
	for(int i=1; i<=n; ++i) {
		if(!b[i]) {
			b[i]=1;
			a[t]=i;
			dfs(t+1);
			b[i]=0;
		}
	}
	return ;
}
void dfss::main () {
	cin>>n;
	dfs(1);
}
